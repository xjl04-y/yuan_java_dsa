天气沙盘 (原数组 temperatures)

备忘录 (Stack 存未来门牌号) top = -1

雷达报告 (res 存等待天数)

运行状态解析

点击右上角的“单步执行代码”开始推演。

public int[] dailyTemperatures(int[] temps) {
int n = temps.length;
int[] stack = new int[n];
int top = -1;
int[] res = new int[n];
for (int i = n - 1; i >= 0; i--) {
while (top != -1 && temps[stack[top]] <= temps[i]) {
top--; // 踢出被遮挡的废数据
}
if (top == -1) {
res[i] = 0; // 右侧无高墙
} else {
res[i] = stack[top] - i; // 🔥 核心：算距离！
}
stack[++top] = i; // 今天入栈，作为过去的未来
}
return res;
}

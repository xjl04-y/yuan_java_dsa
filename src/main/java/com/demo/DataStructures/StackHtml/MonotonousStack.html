public int[] findLeftGreater(int[] nums) {


    int[] res = new int[nums.length];


    for (int i = 0; i < nums.length; i++) {


        // 大清洗：淘汰无效的被遮挡数据


        while (top != -1 && nums[stack[top]] <= nums[i]) {


            top--;


        }


        // 确立历史答案


        if (top == -1) {


            res[i] = -1;


        } else {


            res[i] = stack[top];


        }


        // 新王登基 (注意：存的是索引 i)


        stack[++top] = i;


    }


    return res;


}

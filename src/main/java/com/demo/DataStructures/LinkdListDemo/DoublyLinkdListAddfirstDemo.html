public void addFirst(int data) {
// 1. 创建并初始化新节点
Node newNode = new Node(data, null, null);
if (head == null) {
// 2A. 如果为空，只用单向连接
newNode.next = head;
head = newNode;
size++;
return;
}
// 2B. 不为空，进行双向连接
newNode.next = head;
head.prev = newNode;
head = newNode;
size++;
}

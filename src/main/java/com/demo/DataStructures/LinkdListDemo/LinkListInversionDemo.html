public void Inversion() {
if (head == null) return;
// 初始化变量
Node curr = this.head;
Node prev = null;
// 开始遍历
while (curr != null) {
// 1. 救下未来
Node next = curr.next;
// 2. 现在转身（断开旧线，连新线）
curr.next = prev;
// 3. "过去"走到"现在"的位置
prev = curr;
// 4. "现在"走向"未来"的位置
curr = next;
}
// 收尾，指定新队伍的头
this.head = prev;
}
